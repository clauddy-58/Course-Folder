# --- WAY 1: Using count() with weight argument ---
air.tib |>
  count(Month, wt = Temp) |>
  mutate(
    Month_Char = case_when(
      Month == 5 ~ "May",
      Month == 6 ~ "June",
      Month == 7 ~ "July",
      Month == 8 ~ "August",
      Month == 9 ~ "September"
    ),
    Month_Factor = fct_reorder(Month_Char, n, .desc = TRUE)
  ) |>
  ggplot(aes(x = n, y = Month_Factor)) +
  geom_col(fill = "darkorange") +
  labs(title = "Total Monthly Temperature (Way 1)", x = "Summed Temperature", y = "Month")
# --- WAY 2 (BONUS): Using standard group_by() and summarize() ---
air.tib |>
  group_by(Month) |>
  summarize(total_temp = sum(Temp, na.rm = TRUE)) |>
  mutate(
    Month_Char = month.name[Month],
    Month_Factor = fct_reorder(Month_Char, total_temp, .desc = TRUE)
  ) |>
  ggplot(aes(x = total_temp, y = Month_Factor)) +
  geom_col(fill = "darkred") +
  labs(title = "Total Monthly Temperature (Way 2)", x = "Summed Temperature", y = "Month")
